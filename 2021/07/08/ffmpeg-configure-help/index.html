  --extra-objcflags=FLAGS  #  向OBJCFLAGS添加标志 []
  --extra-ldflags=ELDFLAGS #  将ELDFLAGS添加到LDFLAGS []
  --extra-ldexeflags=ELDFLAGS #  将ELDFLAGS添加到LDEXEFLAGS []
  --extra-ldsoflags=ELDFLAGS #  向LDSOFLAGS添加ELDFLAGS []
  --extra-libs=ELIBS       #  添加ELIBS []
  --extra-version=STRING   #  版本字符串后缀 []
  --optflags=OPTFLAGS      #  重写与优化相关的编译器标志
  --nvccflags=NVCCFLAGS    #  覆盖nvcc标志 []
  --build-suffix=SUFFIX    #  库名称后缀 []
  --enable-pic             #  构建位置独立代码
  --enable-thumb           #  为Thumb指令集编译
  --enable-lto             #  使用链接时间优化
  --env="ENV=override"     #  重写环境变量

Advanced options (experts only): # 高级选项（仅限专家）：
  --malloc-prefix=PREFIX   #  为malloc和相关名称加前缀
  --custom-allocator=NAME  #  使用支持的自定义分配器
  --disable-symver         #  禁用符号版本控制
  --enable-hardcoded-tables #  使用硬编码表而不是运行时生成
  --disable-safe-bitstream-reader #  禁用位读取器中的缓冲区边界检查（速度更快，但可能会崩溃）
  --sws-max-filter-size=N  #  swscale使用的最大过滤器大小 [256]

Optimization options (experts only): # 优化选项（仅限专家）：
  --disable-asm            #  禁用所有程序集优化
  --disable-altivec        #  禁用AltiVec优化
  --disable-vsx            #  禁用VSX优化
  --disable-power8         #  禁用POWER8优化
  --disable-amd3dnow       #  禁用3DNow！优化
  --disable-amd3dnowext    #  禁用3DNow！扩展优化
  --disable-mmx            #  禁用MMX优化
  --disable-mmxext         #  禁用MMXEXT优化
  --disable-sse            #  禁用SSE优化
  --disable-sse2           #  禁用SSE2优化
  --disable-sse3           #  禁用SSE3优化
  --disable-ssse3          #  禁用SSSE3优化
  --disable-sse4           #  禁用SSE4优化
  --disable-sse42          #  禁用SSE4.2优化
  --disable-avx            #  禁用AVX优化
  --disable-xop            #  禁用XOP优化
  --disable-fma3           #  禁用FMA3优化
  --disable-fma4           #  禁用FMA3优化
  --disable-avx2           #  禁用AVX2优化
  --disable-avx512         #  禁用AVX-512优化
  --disable-aesni          #  禁用AESNI优化
  --disable-armv5te        #  禁用armv5te优化
  --disable-armv6          #  禁用armv6优化
  --disable-armv6t2        #  禁用armv6t2优化
  --disable-vfp            #  禁用VFP优化
  --disable-neon           #  禁用NEON优化
  --disable-inline-asm     #  禁止使用内联程序集
  --disable-x86asm         #  禁用独立x86程序集
  --disable-mipsdsp        #  禁用MIPS DSP ASE R1优化
  --disable-mipsdspr2      #  禁用MIPS DSP ASE R2优化
  --disable-msa            #  禁用MSA优化
  --disable-mipsfpu        #  禁用浮点MIPS优化
  --disable-mmi            #  禁用龙芯SIMD优化
  --disable-fast-unaligned #  考虑不对齐的访问慢

Developer options (useful when working on FFmpeg itself): # 开发人员选项（在处理FFmpeg本身时很有用）：
  --disable-debug          #  禁用调试符号
  --enable-debug=LEVEL     #  设置调试级别 []
  --disable-optimizations  #  禁用编译器优化
  --enable-extra-warnings  #  启用更多编译器警告
  --disable-stripping      #  禁用可执行文件和共享库的剥离
  --assert-level=level     #  0（默认），1或2，断言测试量，2导致运行时减速。
  --enable-memory-poisoning #  用任意数据填充堆未初始化的分配空间
  --valgrind=VALGRIND      #  通过valgrind运行“makefate”测试，使用指定的valgrind二进制文件检测内存泄漏和错误。不能与 --target exec组合
  --enable-ftrapv          #  陷阱算术溢出
  --samples=PATH           #  命运测试样本的位置，如果未设置，则在make调用时使用$FATE\u samples。
  --enable-neon-clobber-test #  检查霓虹灯寄存器是否有阻塞（应仅用于调试目的）
  --enable-xmm-clobber-test #  检查XMM寄存器的阻塞（仅限于Win64；应仅用于调试目的）
  --enable-random          #  随机启用/禁用组件
  --disable-random
  --enable-random=LIST     #  随机启用/禁用特定组件
  --disable-random=LIST    #  组件组。LIST是以逗号分隔的NAME[：PROB]条目列表，其中NAME是一个组件（组），PROB是与NAME关联的概率（默认值为0.5）。
  --random-seed=VALUE      #  种子值--启用/禁用随机
  --disable-valgrind-backtrace #  不要在Valgrind下打印回溯（仅适用于--disable optimizations builds）
  --enable-ossfuzz         #  启用building fuzzer工具
  --libfuzzer=PATH         #  libfuzzer的路径
  --ignore-tests=TESTS     #  忽略结果的测试的逗号分隔列表（名称中没有“fate-”前缀）
  --enable-linux-perf      #  启用Linux性能监视器API—禁用大型测试禁用使用大量内存的测试
  --disable-ptx-compression #  即使可能也不要压缩CUDA PTX代码

注意：对象文件是在启动configure的地方生成的。
